module.exports = (sequelize, dataTypes) => {
    let alias = 'Cart_Product';
    let cols = {
        id: {
            type: dataTypes.INTEGER,
            primaryKey: true,
            autoIncrement: true
        },
        cart_id: {
            type: dataTypes.INTEGER
        },
        book_id: {
            type:dataTypes.INTEGER
        }
    };
    let config = {
        tableName: 'cart_products',
        timestamps: false
    };

    const CartProducts = sequelize.define(alias, cols, config);

    CartProducts.associate = function(modelos){

        Cart.belongsToMany(modelos.Books, {
            through: "cart_product",
            foreignKey: 'cart_id',
            otherKey: 'book_id'
        })
    
        Books.belongsToMany(modelos.Cart, {
            through: "cart_product",
            foreignKey: 'book_id',
            otherKey: 'cart_id'
        })

    }


    return CartProducts;
};
